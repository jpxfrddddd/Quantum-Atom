#include "renderer.h"
#include "../physics/orbital.h"
#include <GL/gl.h>
#include <cmath>

void render(float grid[64][64][64]){
    glBegin(GL_POINTS);

    for(float x= -5.0f; x<5.0f; x+=0.1f)
    for(float y= -5.0f; y<5.0f; y+=0.1f)
    for(float z= -5.0f; z<5.0f; z+=0.1f){
        float d = density(x,y,z);
        if(std::isnan(d) || std::isinf(d)) continue; // skip NaN values

        d*=10.0f; // scale density for visibility
        if(d > 1.0f) d = 1.0f;
        if((float)rand() / RAND_MAX < d) // visibility threshold
            glVertex3f(x, y, z);
            glColor3f(d, 0.0f, 1.0f); // color based on density
    }
    glEnd();
}

void initOpenGL(){
    glPointSize(2.0f);
    glEnable(GL_DEPTH_TEST);
}
